Machine Learning Analytics

Implement advanced anomaly detection using machine learning algorithms for behavioral analysis and threat detection.

Anomaly Detection Setup

ml_anomaly_detection.py 
#!/usr/bin/env python3
"""
SIEM Platform ML Anomaly Detection Module
"""

import pandas as pd
import numpy as np
from sklearn.ensemble import IsolationForest
from sklearn.preprocessing import StandardScaler
from elasticsearch import Elasticsearch
import logging

class SIEMAnomalyDetector:
    def __init__(self, es_host="localhost", es_port=9200):
        self.es = Elasticsearch([{'host': es_host, 'port': es_port}])
        self.scaler = StandardScaler()
        self.model = IsolationForest(contamination=0.1, random_state=42)
        
    def fetch_data(self, index_pattern="wazuh-alerts-*", hours=24):
        """Fetch recent security events from Elasticsearch"""
        query = {
            "query": {
                "range": {
                    "@timestamp": {
                        "gte": f"now-{hours}h"
                    }
                }
            },
            "size": 10000
        }
        
        response = self.es.search(index=index_pattern, body=query)
        return response['hits']['hits']
    
    def preprocess_data(self, raw_data):
        """Preprocess security events for ML analysis"""
        features = []
        for event in raw_data:
            source = event['_source']
            feature_vector = [
                source.get('rule', {}).get('level', 0),
                len(source.get('full_log', '')),
                source.get('agent', {}).get('id', 0),
                hash(source.get('rule', {}).get('description', '')) % 1000
            ]
            features.append(feature_vector)
        
        return np.array(features)
    
    def detect_anomalies(self, features):
        """Detect anomalies using Isolation Forest"""
        # Normalize features
        features_scaled = self.scaler.fit_transform(features)
        
        # Train model and predict anomalies
        predictions = self.model.fit_predict(features_scaled)
        anomaly_scores = self.model.decision_function(features_scaled)
        
        return predictions, anomaly_scores
    
    def run_detection(self):
        """Main detection workflow"""
        logging.info("Starting anomaly detection...")
        
        # Fetch and preprocess data
        raw_data = self.fetch_data()
        features = self.preprocess_data(raw_data)
        
        if len(features) == 0:
            logging.warning("No data available for analysis")
            return
        
        # Detect anomalies
        predictions, scores = self.detect_anomalies(features)
        
        # Process results
        anomalies = []
        for i, (pred, score) in enumerate(zip(predictions, scores)):
            if pred == -1:  # Anomaly detected
                anomalies.append({
                    'event_id': raw_data[i]['_id'],
                    'anomaly_score': score,
                    'event_data': raw_data[i]['_source']
                })
        
        logging.info(f"Detected {len(anomalies)} anomalies")
        return anomalies

if __name__ == "__main__":
    detector = SIEMAnomalyDetector()
    anomalies = detector.run_detection()
    
    for anomaly in anomalies:
        print(f"Anomaly detected: {anomaly['event_id']} (score: {anomaly['anomaly_score']:.3f})")

#!/usr/bin/env python3
"""
SIEM Platform Threat Hunting Notebook Integration
"""

import jupyter_client
import nbformat
from nbconvert import HTMLExporter
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from elasticsearch import Elasticsearch

class ThreatHuntingNotebook:
    def __init__(self, es_host="localhost", es_port=9200):
        self.es = Elasticsearch([{'host': es_host, 'port': es_port}])
        
    def create_hunting_notebook(self, hunt_name, description):
        """Create a new threat hunting notebook"""
        nb = nbformat.v4.new_notebook()
        
        # Add title cell
        title_cell = nbformat.v4.new_markdown_cell(f"""
# Threat Hunt: {hunt_name}

**Description:** {description}
**Created:** {pd.Timestamp.now()}
**Status:** Active Investigation

## Hypothesis
[Document your hunting hypothesis here]

## Data Sources
- Wazuh Security Events
- Network Logs
- System Logs
- Threat Intelligence Feeds
        """)
        
        # Add setup cell
        setup_cell = nbformat.v4.new_code_cell("""
# Import required libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from elasticsearch import Elasticsearch
from datetime import datetime, timedelta

# Configure plotting
plt.style.use('seaborn')
sns.set_palette("husl")

# Connect to Elasticsearch
es = Elasticsearch([{'host': 'localhost', 'port': 9200}])

print("Threat hunting environment initialized")
        """)
        
        # Add data collection cell
        data_cell = nbformat.v4.new_code_cell("""
# Define time range for investigation
end_time = datetime.now()
start_time = end_time - timedelta(days=7)

# Query security events
query = {
    "query": {
        "bool": {
            "must": [
                {
                    "range": {
                        "@timestamp": {
                            "gte": start_time.isoformat(),
                            "lte": end_time.isoformat()
                        }
                    }
                }
            ]
        }
    },
    "size": 10000,
    "sort": [{"@timestamp": {"order": "desc"}}]
}

# Execute query
response = es.search(index="wazuh-alerts-*", body=query)
events = [hit['_source'] for hit in response['hits']['hits']]

print(f"Collected {len(events)} security events for analysis")
        """)
        
        # Add analysis cell
        analysis_cell = nbformat.v4.new_code_cell("""
# Convert to DataFrame for analysis
df = pd.json_normalize(events)

# Basic statistics
print("Event Distribution by Rule Level:")
if 'rule.level' in df.columns:
    print(df['rule.level'].value_counts().sort_index())

# Timeline analysis
if '@timestamp' in df.columns:
    df['@timestamp'] = pd.to_datetime(df['@timestamp'])
    df.set_index('@timestamp', inplace=True)
    
    # Plot event timeline
    plt.figure(figsize=(12, 6))
    df.resample('1H').size().plot(kind='line')
    plt.title('Security Events Timeline')
    plt.xlabel('Time')
    plt.ylabel('Event Count')
    plt.xticks(rotation=45)
    plt.tight_layout()
    plt.show()
        """)
        
        # Add cells to notebook
        nb.cells = [title_cell, setup_cell, data_cell, analysis_cell]
        
        # Save notebook
        filename = f"threat_hunt_{hunt_name.lower().replace(' ', '_')}.ipynb"
        with open(filename, 'w') as f:
            nbformat.write(nb, f)
        
        return filename
    
    def execute_notebook(self, notebook_path):
        """Execute a threat hunting notebook"""
        with open(notebook_path, 'r') as f:
            nb = nbformat.read(f, as_version=4)
        
        # Execute notebook
        km = jupyter_client.KernelManager()
        km.start_kernel()
        kc = km.client()
        
        try:
            for cell in nb.cells:
                if cell.cell_type == 'code':
                    kc.execute(cell.source)
                    
        finally:
            km.shutdown_kernel()
        
        return nb
    
    def export_to_html(self, notebook_path, output_path=None):
        """Export notebook to HTML report"""
        with open(notebook_path, 'r') as f:
            nb = nbformat.read(f, as_version=4)
        
        html_exporter = HTMLExporter()
        (body, resources) = html_exporter.from_notebook_node(nb)
        
        if output_path is None:
            output_path = notebook_path.replace('.ipynb', '.html')
        
        with open(output_path, 'w') as f:
            f.write(body)
        
        return output_path

# Example usage
if __name__ == "__main__":
    hunter = ThreatHuntingNotebook()
    
    # Create new hunting notebook
    notebook = hunter.create_hunting_notebook(
        "Lateral Movement Detection",
        "Hunt for signs of lateral movement using authentication logs and network connections"
    )
    
    print(f"Created threat hunting notebook: {notebook}")
    
    # Execute and export
    executed_nb = hunter.execute_notebook(notebook)
    html_report = hunter.export_to_html(notebook)
    
    print(f"Generated HTML report: {html_report}")

class SIEMAPIIntegrator:
    def __init__(self, config_file="api_config.json"):
        with open(config_file, 'r') as f:
            self.config = json.load(f)
        
        self.session = requests.Session()
        self.session.headers.update({
            'Content-Type': 'application/json',
            'User-Agent': 'SIEM-Platform/1.0'
        })
    
    def slack_notification(self, channel: str, message: str, severity: str = "info"):
        """Send notification to Slack"""
        webhook_url = self.config['slack']['webhook_url']
        
        color_map = {
            "info": "#36a64f",
            "warning": "#ff9500", 
            "critical": "#ff0000"
        }
        
        payload = {
            "channel": channel,
            "username": "SIEM Platform",
            "icon_emoji": ":shield:",
            "attachments": [{
                "color": color_map.get(severity, "#36a64f"),
                "title": f"SIEM Alert - {severity.upper()}",
                "text": message,
                "timestamp": int(datetime.now().timestamp())
            }]
        }
        
        try:
            response = self.session.post(webhook_url, json=payload)
            response.raise_for_status()
            logging.info(f"Slack notification sent to {channel}")
            return True
        except Exception as e:
            logging.error(f"Failed to send Slack notification: {e}")
            return False
    
    def servicenow_incident(self, title: str, description: str, priority: int = 3):
        """Create incident in ServiceNow"""
        snow_config = self.config['servicenow']
        url = f"{snow_config['instance_url']}/api/now/table/incident"
        
        auth = (snow_config['username'], snow_config['password'])
        
        payload = {
            "short_description": title,
            "description": description,
            "priority": priority,
            "category": "Security",
            "subcategory": "SIEM Alert",
            "caller_id": snow_config['caller_id'],
            "assignment_group": snow_config['assignment_group']
        }
        
        try:
            response = self.session.post(url, json=payload, auth=auth)
            response.raise_for_status()
            
            incident_data = response.json()
            incident_number = incident_data['result']['number']
            
            logging.info(f"ServiceNow incident created: {incident_number}")
            return incident_number
        except Exception as e:
            logging.error(f"Failed to create ServiceNow incident: {e}")
            return None
    
    def pagerduty_alert(self, title: str, description: str, severity: str = "warning"):
        """Create alert in PagerDuty"""
        pd_config = self.config['pagerduty']
        url = "https://events.pagerduty.com/v2/enqueue"
        
        payload = {
            "routing_key": pd_config['integration_key'],
            "event_action": "trigger",
            "payload": {
                "summary": title,
                "source": "SIEM Platform",
                "severity": severity,
                "component": "Security Monitoring",
                "group": "SOC",
                "class": "Security Alert",
                "custom_details": {
                    "description": description,
                    "timestamp": datetime.now().isoformat()
                }
            }
        }
        
        try:
            response = self.session.post(url, json=payload)
            response.raise_for_status()
            
            result = response.json()
            dedup_key = result.get('dedup_key')
            
            logging.info(f"PagerDuty alert created: {dedup_key}")
            return dedup_key
        except Exception as e:
            logging.error(f"Failed to create PagerDuty alert: {e}")
            return None
    
    def process_security_alert(self, alert_data: Dict):
        """Process security alert and trigger integrations"""
        rule_level = alert_data.get('rule', {}).get('level', 0)
        rule_description = alert_data.get('rule', {}).get('description', 'Unknown')
        agent_name = alert_data.get('agent', {}).get('name', 'Unknown')
        
        title = f"Security Alert: {rule_description}"
        description = f"""
        Alert Details:
        - Rule Level: {rule_level}
        - Agent: {agent_name}
        - Description: {rule_description}
        - Timestamp: {alert_data.get('@timestamp', 'Unknown')}
        
        Full Log: {alert_data.get('full_log', 'N/A')}
        """
        
        # Determine severity and actions based on rule level
        if rule_level >= 12:  # Critical
            severity = "critical"
            
            # Send to all channels for critical alerts
            self.slack_notification("#security-alerts", description, severity)
            self.servicenow_incident(title, description, priority=1)
            self.pagerduty_alert(title, description, severity="critical")
            
        elif rule_level >= 7:  # High
            severity = "warning"
            
            self.slack_notification("#security-alerts", description, severity)
            self.servicenow_incident(title, description, priority=2)
            
        elif rule_level >= 4:  # Medium
            severity = "info"
            
            self.slack_notification("#security-monitoring", description, severity)
        
        logging.info(f"Processed alert with level {rule_level}")
    
    def webhook_handler(self, request_data: Dict):
        """Handle incoming webhook from SIEM components"""
        try:
            # Parse alert data
            if 'alerts' in request_data:
                for alert in request_data['alerts']:
                    self.process_security_alert(alert)
            else:
                self.process_security_alert(request_data)
            
            return {"status": "success", "message": "Alert processed"}
        except Exception as e:
            logging.error(f"Error processing webhook: {e}")
            return {"status": "error", "message": str(e)}
